Order of Day vs Demand per hour

set.seed(1)
Order_hours =
  instacart %>% 
  mutate(order_hour_of_day = as.numeric(order_hour_of_day)) %>% 
  filter(!is.na(order_id))  %>% 
  sample_n(10000) 
Order_hours %>% 
  group_by(order_dow, order_hour_of_day) %>% 
  summarize(demand_per_hour = n()) %>% 
  plot_ly(x = ~ order_hour_of_day, y = ~demand_per_hour, type = "violin", colors = "Set2" )
